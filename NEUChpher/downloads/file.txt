# Calculate greatest common divisor of two integer a and b
def GCD(a, b):  
    if a % b == 0:  
        return b  
    return GCD(b, a % b)  

# Remove non-alphabet characters, such as ( ), (,), (;), etc.
def RemoveNonAlphabet(text):
    newText = ""
    for letter in text:
        if letter.isalpha():
            newText += letter
    return newText
